<script>
  // =============================================================
  //  HELPER FUNCTIONS (declared with `function` so they're hoisted)
  // =============================================================
  function fmt(n, d = 2) {
    if (!isFinite(n)) return '—';
    return n.toLocaleString('en-US', { minimumFractionDigits: d, maximumFractionDigits: d });
  }
  function fmtInt(n) {
    return isFinite(n) ? Math.round(n).toLocaleString('en-US') : '—';
  }
  function fmtCurrency(n) {
    return isFinite(n) ? '฿ ' + Math.round(n).toLocaleString('en-US') : '฿ —';
  }
  function fmtMinSec(minutes) {
    if (!isFinite(minutes) || minutes < 0) return '—';
    const totalSec = Math.round(minutes * 60);
    const m = Math.floor(totalSec / 60);
    const s = totalSec % 60;
    return `${m}:${s.toString().padStart(2, '0')}`;
  }
  function showAlert(type, message) {
    const icon = type === 'warn'
      ? '<svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.2" stroke-linecap="round" stroke-linejoin="round"><path d="M10.29 3.86L1.82 18a2 2 0 0 0 1.71 3h16.94a2 2 0 0 0 1.71-3L13.71 3.86a2 2 0 0 0-3.42 0z"/><line x1="12" y1="9" x2="12" y2="13"/><line x1="12" y1="17" x2="12.01" y2="17"/></svg>'
      : '<svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.2" stroke-linecap="round" stroke-linejoin="round"><circle cx="12" cy="12" r="10"/><line x1="12" y1="16" x2="12" y2="12"/><line x1="12" y1="8" x2="12.01" y2="8"/></svg>';
    document.getElementById('alertBox').innerHTML =
      `<div class="alert alert-${type}">${icon}<span>${message}</span></div>`;
  }
  function clearAlert() {
    document.getElementById('alertBox').innerHTML = '';
  }

  // =============================================================
  //  BOILER ROW MANAGEMENT
  // =============================================================
  let boilerCount = 0;
  const boilerListEl = document.getElementById('boilerList');

  function addBoiler(value = '', labelOverride = null) {
    boilerCount++;
    const row = document.createElement('div');
    row.className = 'boiler-row';
    row.innerHTML = `
      <span class="boiler-label">${labelOverride || ('Boiler ' + (boilerListEl.children.length + 1))}</span>
      <div class="field">
        <input class="input boiler-input" type="number" min="0" step="0.1" value="${value}" placeholder="0.0" />
        <span class="input-unit">t-steam/hr</span>
      </div>
      <button class="remove-btn" title="Remove">
        <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.2" stroke-linecap="round" stroke-linejoin="round"><polyline points="3 6 5 6 21 6"/><path d="M19 6l-1 14a2 2 0 0 1-2 2H8a2 2 0 0 1-2-2L5 6"/><path d="M10 11v6M14 11v6"/></svg>
      </button>
    `;
    boilerListEl.appendChild(row);
    const bInput = row.querySelector('.boiler-input');
    bInput.addEventListener('input', calculate);
    bInput.addEventListener('change', calculate);
    row.querySelector('.remove-btn').addEventListener('click', () => {
      if (boilerListEl.children.length <= 1) return;
      row.remove();
      relabelBoilers();
      calculate();
    });
  }
  function relabelBoilers() {
    [...boilerListEl.children].forEach((row, i) => {
      row.querySelector('.boiler-label').textContent = `Boiler ${i + 1}`;
    });
    [...boilerListEl.children].forEach(r => {
      r.querySelector('.remove-btn').disabled = boilerListEl.children.length <= 1;
    });
  }
  document.getElementById('addBoiler').addEventListener('click', () => {
    addBoiler();
    relabelBoilers();
    calculate();
  });

  // =============================================================
  //  CORE CALCULATION — mirrors the reference Excel
  // =============================================================
  // -------- Fixed engineering constants (matches Excel 3) --------
  // Steam-to-fuel ratio for the reference fuel (bagasse):
  // 1 ton of bagasse produces 2.6 tons of steam.
  const STEAM_TO_FUEL_RATIO = 2.6;
  // Volume of fuel pushed per round-trip (m³). Reference Excel 3:
  // base capacity 773 t/hr at d=3m, density=315 kg/m³, speed=2.4 km/h.
  // Equivalent bucket volume = 773 ÷ (400 trips/hr × 315 kg/m³ ÷ 1000) = 6.135 m³.
  const BUCKET_VOLUME_M3 = 6.135;

  function calculate() {
    // ---- Inputs ----
    const boilerInputs = [...boilerListEl.querySelectorAll('.boiler-input')];
    const totalSteam = boilerInputs.reduce((s, el) => s + (parseFloat(el.value) || 0), 0);

    const distance     = parseFloat(document.getElementById('distance').value) || 0;
    const speedKmh     = parseFloat(document.getElementById('speed').value) || 0;
    const hours        = parseFloat(document.getElementById('hours').value) || 0;
    const costRate     = parseFloat(document.getElementById('costRate').value) || 0;

    const propBagasse  = parseFloat(document.getElementById('propBagasse').value) || 0;
    const lhvBagasse   = parseFloat(document.getElementById('lhvBagasse').value) || 0;
    const densBagasse  = parseFloat(document.getElementById('densBagasse').value) || 0;
    const propLeaves   = parseFloat(document.getElementById('propLeaves').value) || 0;
    const lhvLeaves    = parseFloat(document.getElementById('lhvLeaves').value) || 0;
    const densLeaves   = parseFloat(document.getElementById('densLeaves').value) || 0;

    // ---- Boiler total label ----
    document.getElementById('boilerSum').textContent =
      `Total: ${fmt(totalSteam, 2)} t-steam/hr`;

    // ---- Fuel proportions (Excel uses raw %, NOT normalized) ----
    const propSum = propBagasse + propLeaves;
    document.getElementById('fuelTotal').textContent = `Total: ${propSum}%`;

    // ---- Round-trip time (Excel section 1) ----
    // Excel E6:  (distance × 2) ÷ (speed_km/h × 1000 / 60)  →  minutes
    const speedMpm = (speedKmh * 1000) / 60;                              // m/min
    const tripTime = speedMpm > 0 ? (2 * distance) / speedMpm : Infinity; // minutes
    const tripsPerHour = tripTime > 0 ? 60 / tripTime : 0;

    // ---- Weight ratio (Excel section 2, column D) ----
    // ratio_i = LHV_i ÷ LHV_bagasse (bagasse is reference, ratio = 1)
    const ratioBagasse = 1;
    const ratioLeaves  = lhvBagasse > 0 ? lhvLeaves / lhvBagasse : 0;

    // ---- Fuel weight required (Excel section 2, column E) ----
    // Excel: fuel_i = (Steam ÷ 2.6) × (prop_i ÷ 100) ÷ (LHV_i ÷ LHV_bagasse)
    // Equivalent: fuel_i = Steam × (prop_i/100) × LHV_bagasse ÷ (2.6 × LHV_i)
    const fuelBagasse = (lhvBagasse > 0)
      ? (totalSteam / STEAM_TO_FUEL_RATIO) * (propBagasse / 100) / ratioBagasse
      : 0;
    const fuelLeaves  = (lhvLeaves > 0 && ratioLeaves > 0)
      ? (totalSteam / STEAM_TO_FUEL_RATIO) * (propLeaves / 100) / ratioLeaves
      : 0;
    const fuelTotalAmt = fuelBagasse + fuelLeaves;

    // ---- Vehicle capacity per fuel (Excel section 3) ----
    // capacity_i (t/hr) = trips/hr × bucket_volume × density_i ÷ 1000
    const capBagasse = tripsPerHour * BUCKET_VOLUME_M3 * densBagasse / 1000;
    const capLeaves  = tripsPerHour * BUCKET_VOLUME_M3 * densLeaves  / 1000;

    // ---- Vehicles needed (Excel section 4) ----
    // sum of vehicle fractions per fuel, then CEIL
    let fraction = 0;
    if (capBagasse > 0) fraction += fuelBagasse / capBagasse;
    if (capLeaves  > 0) fraction += fuelLeaves  / capLeaves;
    const vehicles = (totalSteam > 0 && fraction > 0) ? Math.ceil(fraction) : 0;
    const utilization = vehicles > 0 ? (fraction / vehicles) * 100 : 0;

    // ---- Daily cost (Excel section 5) ----
    const dailyCost = vehicles * hours * costRate;

    // =========== RENDER ===========
    document.getElementById('totalDemand').textContent  = fmt(totalSteam, 2);
    document.getElementById('tripTimeFmt').textContent  = fmtMinSec(tripTime);
    document.getElementById('trips').textContent        = fmt(tripsPerHour, 1);

    document.getElementById('fuelBagasse').textContent  = fmt(fuelBagasse, 2);
    document.getElementById('fuelLeaves').textContent   = fmt(fuelLeaves, 2);
    document.getElementById('fuelTotalAmt').textContent = fmt(fuelTotalAmt, 2);

    document.getElementById('capBagasse').textContent   = fmt(capBagasse, 2);
    document.getElementById('capLeaves').textContent    = fmt(capLeaves, 2);
    document.getElementById('utilization').textContent  = fmt(utilization, 1);

    document.getElementById('vehiclesValue').textContent = fmtInt(vehicles);
    document.getElementById('vehiclesUnit').textContent  =
      vehicles === 1 ? 'unit' : 'units';
    document.getElementById('dailyCost').textContent = fmtCurrency(dailyCost);

    // ---- Alerts ----
    clearAlert();
    if (totalSteam <= 0) {
      showAlert('info', 'Enter at least one boiler load to begin · กรุณาใส่ค่า Boiler load');
    } else if (propSum !== 100) {
      showAlert('warn', `Fuel proportions sum to ${propSum}% — they should total 100% for accurate fuel weight.`);
    } else if (!isFinite(tripTime) || tripTime === 0) {
      showAlert('warn', 'Vehicle speed and distance must be greater than zero.');
    } else if (vehicles >= 5) {
      showAlert('warn', `High vehicle requirement (${vehicles} units). Consider reducing distance or increasing speed.`);
    } else {
      showAlert('info', `Fleet of ${vehicles} vehicle${vehicles>1?'s':''} runs at ${fmt(utilization,1)}% utilization — handling ${fmt(fuelTotalAmt,2)} t/hr of fuel.`);
    }
  }

  // =============================================================
  //  INITIALIZATION (runs once, after every function is defined)
  // =============================================================

  // Seed with the boilers shown in the reference Excel
  addBoiler(120, 'Boiler 11');
  addBoiler(120, 'Boiler 12');
  addBoiler(170, 'Boiler 21');
  // Re-label them 1, 2, 3 so the UI stays consistent if user adds/removes
  relabelBoilers();

  // Wire up all parameter inputs
  ['distance','speed','hours','costRate',
   'propBagasse','lhvBagasse','densBagasse',
   'propLeaves','lhvLeaves','densLeaves'].forEach(id => {
    const el = document.getElementById(id);
    el.addEventListener('input', calculate);
    el.addEventListener('change', calculate);
  });

  // Calculate / Reset buttons
  const calcBtn   = document.getElementById('calcBtn');
  const resetBtn  = document.getElementById('resetBtn');
  const resultCard = document.querySelector('.result-card');

  calcBtn.addEventListener('click', () => {
    calculate();
    calcBtn.classList.remove('flash');
    void calcBtn.offsetWidth;
    calcBtn.classList.add('flash');
    resultCard.classList.remove('updated');
    void resultCard.offsetWidth;
    resultCard.classList.add('updated');
    if (window.innerWidth < 980) {
      resultCard.scrollIntoView({ behavior: 'smooth', block: 'start' });
    }
  });

  resetBtn.addEventListener('click', () => {
    document.getElementById('distance').value     = 60;
    document.getElementById('speed').value        = 2.4;
    document.getElementById('hours').value        = 24;
    document.getElementById('costRate').value     = 1313;
    document.getElementById('propBagasse').value  = 60;
    document.getElementById('lhvBagasse').value   = 7700;
    document.getElementById('densBagasse').value  = 315;
    document.getElementById('propLeaves').value   = 40;
    document.getElementById('lhvLeaves').value    = 12000;
    document.getElementById('densLeaves').value   = 80;
    boilerListEl.innerHTML = '';
    addBoiler(120);
    addBoiler(120);
    addBoiler(170);
    relabelBoilers();
    calculate();
  });

  // Press Enter inside any input → Calculate
  document.querySelectorAll('.input').forEach(inp => {
    inp.addEventListener('keydown', e => {
      if (e.key === 'Enter') { e.preventDefault(); calcBtn.click(); }
    });
  });

  // Fuel type toggle (Compacted / Uncompacted)
  // Updates the density input to the preset value, then recalculates.
  document.querySelectorAll('.fuel-type-toggle').forEach(toggle => {
    toggle.addEventListener('click', e => {
      const btn = e.target.closest('button');
      if (!btn || btn.classList.contains('active')) return;
      // swap active state
      toggle.querySelectorAll('button').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      // auto-fill the density input
      const targetId = toggle.dataset.target;
      const density  = btn.dataset.density;
      const densInput = document.getElementById(targetId);
      if (densInput && density != null) {
        densInput.value = density;
      }
      calculate();
    });
  });

  // First render
  calculate();
</script>
